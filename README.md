## Installation

```
npm install --save react@latest
npm install --save @trendmicro/react-anchor
```

```js
import Anchor from '@trendmicro/react-anchor';
```

## Usage

```js
<Anchor
    className="btn btn-default"
    href="https://github.com"
>
    <i className="fa fa-github fa-fw" />
    GitHub
</Anchor>
```
